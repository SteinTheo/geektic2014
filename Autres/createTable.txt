create table geek (id_geek int identity primary key, nom varchar(50), prenom varchar(50), mail varchar(100), sexe char(1), avatar varchar(255));

create table interet (id_int int identity primary key, nom varchar(100), type varchar(50));

create table interet_geek (id_geek int not null foreign key references geek(id_geek), id_interet int not null foreign key references interet(id_int)) ;
